        // --- Database Setup and Authentication ---

        async function initializeFirebase() {

            try {

                const app = initializeApp(firebaseConfig);

                auth = getAuth(app);

                db = getFirestore(app);

               

                const collectionPath = `/artifacts/${appId}/public/data/arcraiders_trades`;

                tradesCollectionRef = collection(db, collectionPath);

                if (initialAuthToken) {

                    await signInWithCustomToken(auth, initialAuthToken);

                } else {

                    await signInAnonymously(auth);

                }

               

                onAuthStateChanged(auth, (user) => {

                    if (user) {

                        userId = user.uid;

                        document.getElementById('post-form').classList.remove('hidden');

                        document.getElementById('user-id-display').textContent = `Your ID: ${userId}`;

                    } else {

                        userId = null;

                    }

                    isAuthReady = true;

                    // Start on the live feed

                    switchView('live');

                });

            } catch (e) {

                console.error("Error initializing Firebase:", e);

                showMessage("Failed to connect to the database. Check console for details.", true);

            }

        }

        // --- Trade Management ---

        // Listener for Pending Trades (Live Feed)

        function setupLiveListener() {

            if (!isAuthReady || !db) return;

            // Query for trades with status 'pending', ordered by timestamp descending

            const q = query(tradesCollectionRef, where("status", "==", "pending"), orderBy("timestamp", "desc"));

            onSnapshot(q, (snapshot) => {

                const trades = [];

                snapshot.forEach(doc => {

                    trades.push({ id: doc.id, ...doc.data() });

                });

                renderLiveTrades(trades);

            }, (error) => {

                console.error("Error listening to live trades:", error);

                showMessage("Error fetching live trades. Please refresh.", true);

            });

        }

       

        // Listener for Accepted Trades (Trade Archive)

        function setupHistoryListener() {

            if (!isAuthReady || !db) return;

            // Query for trades with status 'accepted', ordered by acceptedAt descending (most recent first)

            const q = query(tradesCollectionRef, where("status", "==", "accepted"), orderBy("acceptedAt", "desc"));

            onSnapshot(q, (snapshot) => {

                const trades = [];

                snapshot.forEach(doc => {

                    trades.push({ id: doc.id, ...doc.data() });

                });

                renderHistoryList(trades);

            }, (error) => {

                console.error("Error listening to trade history:", error);

                showMessage("Error fetching trade history. Please refresh.", true);

            });

        }

        async function postNewTrade(event) {

            event.preventDefault();

            if (!userId) {

                showMessage("Authentication not ready. Please wait a moment.", true);

                return;

            }

            const form = event.target;

            const givingItem = form.elements['givingItem'].value;

            const givingQuantity = parseInt(form.elements['givingQuantity'].value);

            const seekingItem = form.elements['seekingItem'].value;

            const seekingQuantity = parseInt(form.elements['seekingQuantity'].value);

            const platform = form.elements['platform'].value;

            const gamerTag = form.elements['gamerTag'].value.trim();

            if (!givingItem || isNaN(givingQuantity) || givingQuantity <= 0 || !seekingItem || isNaN(seekingQuantity) || seekingQuantity <= 0 || !platform || !gamerTag) {

                showMessage("Please fill out all trade details correctly.", true);

                return;

            }

            try {

                await addDoc(tradesCollectionRef, {

                    userId: userId,

                    platform: platform,

                    gamerTag: gamerTag,

                    givingItem: givingItem,

                    givingQuantity: givingQuantity,

                    seekingItem: seekingItem,

                    seekingQuantity: seekingQuantity,

                    status: 'pending', // New status field

                    timestamp: Timestamp.now()

                });

                form.reset();

                form.elements['givingItem'].value = "";

                form.elements['seekingItem'].value = "";

                showMessage("Trade posted successfully!");

            } catch (e) {

                console.error("Error adding document: ", e);

                showMessage("Failed to post trade. Check console for error.", true);

            }

        }

        async function deleteTrade(tradeId) {

            if (!userId) return;

            try {

                const tradeDocRef = doc(db, tradesCollectionRef.path, tradeId);

                await deleteDoc(tradeDocRef);

                showMessage("Trade removed.");

            } catch (e) {

                console.error("Error removing document: ", e);

                showMessage("Failed to remove trade. Only the poster can delete it.", true);

            }

        }

        // Global function to trigger the completion process (after modal confirmation)

        async function confirmCompletion() {

            if (!userId || !tradeToComplete) {

                showMessage("Invalid trade or user state.", true);

                return;

            }

            try {

                const tradeDocRef = doc(db, tradesCollectionRef.path, tradeToComplete);

                await updateDoc(tradeDocRef, {

                    status: 'accepted',

                    acceptedAt: Timestamp.now() // Record when the trade was completed

                });

               

                hideCompletionModal();

                showMessage("Trade marked as completed and archived!", false);

            } catch (e) {

                console.error("Error completing document: ", e);

                showMessage("Failed to complete trade. Check console for error.", true);

            }

        }

        }
